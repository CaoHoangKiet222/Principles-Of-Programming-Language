
      lengthOfLastWord: function int(s: array[100] of string, size: int) {
        count: int = 0;
        if (size == 0)
          return 0;

        i: int = size - 1,
        while ((s[i] == " ") && i >= 0) {
          i = i - 1;
        }
        while ((i >= 0) && s[i] != " ") {
          count = count + 1;
          i = i - 1;
        }
        return count;
      }
