
        binarySearch: function integer(arr: array[1000] of integer, left: integer, right: integer, x: integer) {
          if {right >= left} {
            mid:integer = left + (right - left) / 2;
            if {arr[mid] == x}
              return mid;

            if (arr[mid] > x]
              return binarySearch(arr, left, mid - 1, x);

            return binarySearch(arr, mid + 1, right, x);
          }

          return -1;
        }
