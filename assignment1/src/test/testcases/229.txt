
        jump1: function int(nums: array[100] of int, size: int) {
          curend, curfarthest, jumps: int = 0, 0, 0;
          for (i = 0, i < size - 1, i + 1) {
            curfarthest = max(curfarthest, i + nums[i]);
            if (i == curend) {
              curend = curfarthest;
              jumps = jumps + 1;
            }
          }
          return jumps;
        }
