
      search: function int(nums: array[100] of int, size: int, target: int) {
        left, right: int = 0, size - 1;
        index: int = -1;
        found: boolean = false;
        while (left <= right && !found) {
          mid: int = (left + right) / 2;
          if (nums[mid] == target) {
            found = true;
            index = mid;
          }
          if (nums[mid] >= nums[left]) {
            if (nums[mid] > target) {
              if (target < nums[left])
                left = mid + 1;
              else
                right = mid - 1;
            } else
              left = mid + 1;
          } else {
            if (target > nums[mid]) {
              if (target > nums[right])
                right = mid - 1;
              else
                left = mid + 1;
            } else
              right = mid - 1;
          }
        }
        return index;
      }
