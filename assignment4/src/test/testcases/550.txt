
            inc: function float(n: integer, delta: integer) {
                n = n + delta;
                for (i = 1, i < n, i+1) {
                    for (j = 1, j < n, j+1) {
                        if (i + j >= 5) {
                            return i + j;
                        } else {
                            printInteger(i-j);
                        }
                    }
                }
                return 10;
            }

                writeFloat(inc(2, 2));
