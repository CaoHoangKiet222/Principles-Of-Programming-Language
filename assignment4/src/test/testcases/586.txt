
        findMin: function integer(vals: array[100] of integer, numEls: integer) {
          min: integer = vals[0];
          for (i = 1, i < numEls, i+1) {
            if (vals[i] < min) {
              min = vals[i];
            }
          }
          return min;
        }

        main: function void() {
            arr: array[10] of integer = {10, 2, 3, 4};
            n: integer = 10;
            printInteger(findMin(arr, n));
        }
