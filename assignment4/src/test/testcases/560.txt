
        less_zero: boolean = false;
        c: integer = 0;

        printPattern: function void(n: integer) {
          if (n <= 0) {
            less_zero = true;
          }

          if (less_zero) {
            c = c - 1;
            if (c == -1) {
                printInteger(n);
                return;
            }
            printPattern(n + 5);
          } else {
            c = c + 1;
            printPattern(n - 5);
          }
        }

            printPattern(5);
